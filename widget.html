    <script>

        const employees = ["Morris", "Jojo", "Jan", "Piet", "Philly", "Darija"];

        let records = JSON.parse(localStorage.getItem("workRecords")) || {};


        function getCurrentTime() {

            return new Date().toLocaleTimeString();

        }


        function updateTable() {

            const tableBody = document.getElementById("table-body");

            tableBody.innerHTML = "";


            employees.forEach(name => {

                let userData = records[name] || {};

                tableBody.innerHTML += `

                    <tr>

                        <td>${name}</td>

                        <td><button class="start" onclick="recordTime('${name}', 'start')">Start</button></td>

                        <td><button class="pause" onclick="recordTime('${name}', 'pause')">Pause</button></td>

                        <td><button class="pause" onclick="recordTime('${name}', 'pauseOff')">Pause Off</button></td>

                        <td><button class="end" onclick="recordTime('${name}', 'end')">Ende</button></td>

                    </tr>

                `;

            });

        }


        function recordTime(name, action) {

            if (!records[name]) records[name] = {};

            records[name][action] = getCurrentTime();

            localStorage.setItem("workRecords", JSON.stringify(records));

            updateTable();

        }


        updateTable();

    </script>
